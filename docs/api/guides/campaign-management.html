Campaign Management Guide

Learn how to create and manage marketing campaigns using the Ultimate Marketing Team API.

Back to Documentation Home

Overview

The Campaign API allows you to create, manage, and analyze marketing campaigns across multiple channels. This guide covers common campaign management workflows and best practices.

Note: All campaign endpoints require authentication. Make sure to include your JWT token in the Authorization header.

List Campaigns

Retrieve a list of campaigns with optional filtering:

Endpoint: GET /api/v1/campaigns GET

Query Parameters:

  • brand_id: Filter by brand ID
  • status: Filter by status (draft, scheduled, active, paused, ended, archived)
  • start_date: Filter campaigns starting on or after this date
  • end_date: Filter campaigns ending on or before this date
  • page: Page number (default: 1)
  • limit: Results per page (default: 20, max: 100)

Code Examples

Python - Create Campaign and Track Performance

import requests
import json
from datetime import datetime, timedelta

# Authentication (replace with your token)
token = "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9..."
headers = {
    "Authorization": f"Bearer {token}",
    "Content-Type": "application/json"
}
base_url = "https://api.ultimatemarketingteam.com/api/v1"

# Step 1: Create a campaign
campaign_data = {
    "name": "Winter Holiday Campaign",
    "description": "Promoting winter product line during holiday season",
    "brand_id": "123e4567-e89b-12d3-a456-426614174000",
    "budget": 10000.00,
    "currency": "USD",
    "start_date": "2025-11-15T00:00:00Z",
    "end_date": "2025-12-31T23:59:59Z",
    "objective": "conversions",
    "status": "draft",
    "tags": ["winter", "holiday", "sales"]
}

campaign_response = requests.post(
    f"{base_url}/campaigns",
    headers=headers,
    json=campaign_data
)

campaign = campaign_response.json()
campaign_id = campaign["id"]

# Function to check campaign performance
def check_campaign_performance(campaign_id, days=7):
    # Calculate date range
    end_date = datetime.now().strftime("%Y-%m-%d")
    start_date = (datetime.now() - timedelta(days=days)).strftime("%Y-%m-%d")
    
    # Get campaign metrics
    metrics_response = requests.get(
        f"{base_url}/campaigns/{campaign_id}/metrics",
        headers=headers,
        params={"start_date": start_date, "end_date": end_date}
    )
    
    if metrics_response.status_code == 200:
        metrics = metrics_response.json()
        
        # Print summary
        summary = metrics["summary"]
        print(f"\nCampaign Performance Summary ({days} days):")
        print(f"Impressions: {summary['impressions']:,}")
        print(f"Clicks: {summary['clicks']:,}")
        print(f"CTR: {summary['ctr']:.2%}")
        print(f"Spend: ${summary['spend']:.2f}")
        print(f"Conversions: {summary['conversions']:,}")
        print(f"Cost per conversion: ${summary['cost_per_conversion']:.2f}")
        print(f"Conversion rate: {summary['conversion_rate']:.2%}")
        
        # Print platform breakdown
        print("\nPerformance by Platform:")
        for platform in metrics["by_platform"]:
            print(f"- {platform['platform'].capitalize()}: {platform['impressions']:,} impressions, {platform['clicks']:,} clicks, ${platform['spend']:.2f} spend")
        
        return metrics
    else:
        print(f"Error retrieving metrics: {metrics_response.status_code}")
        return None

// Authentication (replace with your token)
const token = "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...";
const headers = {
  "Authorization": `Bearer ${token}`,
  "Content-Type": "application/json"
};
const baseUrl = "https://api.ultimatemarketingteam.com/api/v1";

// Function to create an A/B test
async function createAbTest(campaignId, adSetId, variantAId, variantBId) {
  try {
    const testData = {
      name: "CTA Button A/B Test",
      description: "Testing different call-to-action buttons",
      ad_set_id: adSetId,
      variants: [
        {
          ad_id: variantAId,
          name: "Shop Now",
          weight: 50
        },
        {
          ad_id: variantBId,
          name: "Learn More",
          weight: 50
        }
      ],
      success_metric: "conversion_rate",
      duration_days: 14,
      minimum_confidence: 95
    };
    
    const response = await fetch(`${baseUrl}/campaigns/${campaignId}/ab-tests`, {
      method: "POST",
      headers: headers,
      body: JSON.stringify(testData)
    });
    
    if (!response.ok) {
      throw new Error(`API error: ${response.status}`);
    }
    
    const abTest = await response.json();
    console.log(`Created A/B test: ${abTest.id}`);
    return abTest;
  } catch (error) {
    console.error("Error creating A/B test:", error);
    return null;
  }
}

// Function to check A/B test results
async function checkAbTestResults(campaignId, testId) {
  try {
    const response = await fetch(`${baseUrl}/campaigns/${campaignId}/ab-tests/${testId}/results`, {
      method: "GET",
      headers: headers
    });
    
    if (!response.ok) {
      throw new Error(`API error: ${response.status}`);
    }
    
    const results = await response.json();
    
    // Output results
    console.log(`\nA/B Test Results: ${results.name}`);
    console.log(`Status: ${results.status}`);
    
    if (results.status === "completed") {
      const winner = results.results.winner;
      
      console.log(`\nWinner: ${winner.variant_name}`);
      console.log(`Improvement: ${winner.improvement.toFixed(1)}%`);
      console.log(`Confidence: ${winner.confidence.toFixed(1)}%`);
      
      console.log("\nVariant Results:");
      results.results.variants.forEach(variant => {
        console.log(`- ${variant.name}:`);
        console.log(`  Impressions: ${variant.impressions.toLocaleString()}`);
        console.log(`  Clicks: ${variant.clicks.toLocaleString()}`);
        console.log(`  CTR: ${(variant.click_through_rate * 100).toFixed(2)}%`);
        console.log(`  Conversions: ${variant.conversions.toLocaleString()}`);
        console.log(`  Conversion Rate: ${(variant.conversion_rate * 100).toFixed(2)}%`);
      });
      
      console.log(`\nRecommendation: ${results.recommendation}`);
    } else {
      console.log("Test is still running or hasn't collected enough data.");
    }
    
    return results;
  } catch (error) {
    console.error("Error checking A/B test results:", error);
    return null;
  }
}

// Function to apply A/B test winner
async function applyAbTestWinner(campaignId, testId) {
  try {
    const response = await fetch(`${baseUrl}/campaigns/${campaignId}/ab-tests/${testId}/apply-winner`, {
      method: "POST",
      headers: headers
    });
    
    if (!response.ok) {
      throw new Error(`API error: ${response.status}`);
    }
    
    const result = await response.json();
    console.log(`Applied winning variant: ${result.winner_variant}`);
    console.log(`Updated ad: ${result.updated_ad_id}`);
    return result;
  } catch (error) {
    console.error("Error applying A/B test winner:", error);
    return null;
  }
}

Best Practices

  • Always clearly define campaign objectives before creating campaigns.
  • Set appropriate budget constraints to avoid overspending.
  • Use A/B testing to optimize ad creative and messaging.
  • Segment audiences carefully for better targeting precision.
  • Monitor campaign performance regularly and make adjustments as needed.
  • Use campaign analytics to inform future marketing strategies.
  • Create specific ad sets for different audience segments or platforms.
  • Link relevant content to campaigns for consistent messaging.
  • Take advantage of platform-specific features for each ad platform.
  • Test different ad formats (image, video, carousel) to determine what works best for your audience.
Tip: For optimal performance, create multiple ad variations and use A/B testing to determine which ones resonate most with your audience. Then allocate more budget to the top-performing ads.
